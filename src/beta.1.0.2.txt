// ============================================================
// Software ADC Filter (replaces hardware bypass capacitor)
// ============================================================
// 1) EMA (Exponential Moving Average) — mimics RC low-pass filter

// ============================================================
// Modbus RTU helpers for CWT Soil Sensor (RS485)
// Register map (read 4 from 0x0000):
//   0x00 Humidity  (x0.1 %RH)
//   0x01 Temp      (x0.1 degC)
//   0x02 EC        (1 uS/cm)
//   0x03 pH        (x0.1)
// ============================================================
#define SOIL_DEVICE_ADDR 0x01

uint16_t modbusCalcCRC(uint8_t* buf, uint8_t len) {
  uint16_t crc = 0xFFFF;
  for (uint8_t i = 0; i < len; i++) {
    crc ^= (uint16_t)buf[i];
    for (uint8_t j = 0; j < 8; j++) {
      if (crc & 0x0001) { crc >>= 1; crc ^= 0xA001; }
      else               { crc >>= 1; }
    }
  }
  return crc;
}

bool soilReadRegisters(uint16_t regAddr, uint8_t count, uint16_t* values) {
  while (SoilSerial.available()) SoilSerial.read(); // flush stale bytes

  uint8_t request[8];
  request[0] = SOIL_DEVICE_ADDR;
  request[1] = 0x03;                     // Read Holding Registers
  request[2] = (regAddr >> 8) & 0xFF;
  request[3] =  regAddr       & 0xFF;
  request[4] = 0x00;
  request[5] = count;
  uint16_t crc = modbusCalcCRC(request, 6);
  request[6] = crc & 0xFF;
  request[7] = (crc >> 8) & 0xFF;

  SoilSerial.write(request, 8);
  SoilSerial.flush();

  // Expected: addr(1)+fc(1)+bytecount(1)+data(count*2)+crc(2)
  uint8_t expectedLen = 5 + count * 2;
  uint8_t response[64];
  unsigned long t0 = millis();
  uint8_t idx = 0;
  while (millis() - t0 < 500 && idx < expectedLen) {
    if (SoilSerial.available())
      response[idx++] = (uint8_t)SoilSerial.read();
  }

  if (idx < expectedLen) {
    Serial.printf("[Soil] Timeout: got %d/%d bytes\n", idx, expectedLen);
    return false;
  }

  uint16_t recvCRC = (uint16_t)response[idx-2] | ((uint16_t)response[idx-1] << 8);
  if (recvCRC != modbusCalcCRC(response, idx - 2)) {
    Serial.println("[Soil] CRC mismatch");
    return false;
  }
  if (response[0] != SOIL_DEVICE_ADDR || response[1] != 0x03) {
    Serial.println("[Soil] Bad response header");
    return false;
  }

  for (uint8_t i = 0; i < count; i++)
    values[i] = ((uint16_t)response[3 + i*2] << 8) | response[4 + i*2];

  return true;
}

// ============================================================
// Sensor Read Functions
// ============================================================
WaterLevelData readWaterLevel() {
  WaterLevelData result = {0, 0, false};

  // ── FIX 1: WiFi modem sleep during ADC burst ──────────────────────────
  // Pause WiFi radio transmissions for the ~50ms sampling window so RF
  // bursts don't inject noise into the shared 3.3V ADC rail.
  wifi_ps_type_t prevPS;
  esp_wifi_get_ps(&prevPS);
  esp_wifi_set_ps(WIFI_PS_MAX_MODEM);

  // Oversample ADC: 1000 samples + sort + discard outliers
  int adcBuf[ADC_SAMPLES];
  for (int i = 0; i < ADC_SAMPLES; i++) adcBuf[i] = analogRead(WATER_LEVEL_ADC_PIN);

  // Restore WiFi power save mode immediately after sampling
  esp_wifi_set_ps(prevPS);
  // ──────────────────────────────────────────────────────────────────────

  // Insertion sort then trimmed mean (discard top+bottom ADC_DISCARD outliers)
  for (int i = 1; i < ADC_SAMPLES; i++) {
    int key = adcBuf[i], j = i - 1;
    while (j >= 0 && adcBuf[j] > key) { adcBuf[j+1] = adcBuf[j]; j--; }
    adcBuf[j+1] = key;
  }
  long adcSum  = 0;
  int adcCount = ADC_SAMPLES - 2 * ADC_DISCARD;
  for (int i = ADC_DISCARD; i < ADC_SAMPLES - ADC_DISCARD; i++) adcSum += adcBuf[i];
  int adcRaw = (int)(adcSum / adcCount);

  // Convert ADC → voltage → mA
  float voltage = (adcRaw / 4095.0f) * 3.3f;
  float mA      = (voltage / SHUNT_OHMS) * 1000.0f;
  currentMA     = mA;

  // Fault detection: below 3.5mA = open circuit / sensor disconnected
  if (mA < 3.5f) {
    Serial.printf("[Water] ADC=%d  %.2fmA — below 4mA, sensor open-circuit?\n", adcRaw, mA);
    return result;  // isValid = false
  }

  // Linear mapping: zeroMA = 0 cm, 20mA = 500 cm
  float rawDepth = ((mA - zeroMA) / (20.0f - zeroMA)) * 500.0f;
  rawDepth = max(rawDepth, 0.0f);

  // ── FIX 2: Spike rejection (rate-of-change clamp) ─────────────────────
  // If the reading jumps more than SPIKE_MAX_DELTA_CM from the last accepted
  // value it is treated as a transient spike. It must persist for
  // SPIKE_CONFIRM_CYCLES consecutive readings before being accepted as real.
  float filteredDepth;
  if (lastAcceptedDepth < -9000.0f) {
    // First ever reading — seed everything and accept unconditionally
    lastAcceptedDepth = rawDepth;
    pendingDepth      = rawDepth;
    spikeCounter      = 0;
    filteredDepth     = rawDepth;
    Serial.printf("[Water] Filter seeded: %.1fcm\n", rawDepth);
  } else {
    float delta = fabsf(rawDepth - lastAcceptedDepth);
    if (delta <= SPIKE_MAX_DELTA_CM) {
      // Normal change — accept immediately, reset spike counter
      lastAcceptedDepth = rawDepth;
      pendingDepth      = rawDepth;
      spikeCounter      = 0;
      filteredDepth     = rawDepth;
    } else {
      // Suspicious jump — start or continue confirmation window
      if (fabsf(rawDepth - pendingDepth) <= SPIKE_MAX_DELTA_CM) {
        spikeCounter++;
      } else {
        // New spike candidate — reset confirmation
        pendingDepth = rawDepth;
        spikeCounter = 1;
      }

      if (spikeCounter >= SPIKE_CONFIRM_CYCLES) {
        // Persisted long enough — it's a real level change, accept it
        Serial.printf("[Water] Spike confirmed after %d cycles: %.1fcm->%.1fcm\n",
                      spikeCounter, lastAcceptedDepth, rawDepth);
        lastAcceptedDepth = rawDepth;
        spikeCounter      = 0;
        filteredDepth     = rawDepth;
      } else {
        // Still looks like a transient — hold last accepted value
        Serial.printf("[Water] Spike REJECTED (delta=%.1fcm, cycle %d/%d) raw=%.1fcm held=%.1fcm\n",
                      delta, spikeCounter, SPIKE_CONFIRM_CYCLES, rawDepth, lastAcceptedDepth);
        filteredDepth = lastAcceptedDepth;
      }
    }
  }
  // ──────────────────────────────────────────────────────────────────────

  // ── FIX 3: EMA low-pass filter (software capacitor) ───────────────────
  // Applies exponential smoothing on the spike-cleaned value.
  // alpha=0.1 → output moves 10% toward new reading each cycle (heavy smooth)
  if (emaDepth < -9000.0f) {
    emaDepth = filteredDepth;  // seed on first valid reading
  } else {
    emaDepth = EMA_ALPHA * filteredDepth + (1.0f - EMA_ALPHA) * emaDepth;
  }
  // ──────────────────────────────────────────────────────────────────────

  distance_cm          = emaDepth;
  result.sensorReading = emaDepth;
  result.waterDepth    = emaDepth - waterOffset;
  result.isValid       = true;

  Serial.printf("[Water] ADC=%d  %.3fmA  raw=%.1fcm  filtered=%.1fcm  ema=%.1fcm  waterDepth=%.1fcm\n",
                adcRaw, mA, rawDepth, filteredDepth, emaDepth, result.waterDepth);
  return result;
}

void readSoilSensor() {
  uint16_t regs[4];
  if (soilReadRegisters(0x0000, 4, regs)) {
    // Apply calibration offsets
    soilData.humidity     = (regs[0] / 10.0f) + calHumidOffset;
    soilData.temperature  = (regs[1] / 10.0f) + calTempOffset;
    soilData.conductivity = (float)regs[2]     + calEcOffset;
    soilData.ph           = (regs[3] / 10.0f) + calPhOffset;
    soilData.valid        = true;
    Serial.printf("[Soil] T=%.1fC H=%.1f%% EC=%.0f pH=%.1f\n",
      soilData.temperature, soilData.humidity, soilData.conductivity, soilData.ph);
  } else {
    soilData.valid = false;
  }
}

void readRainfallSensor() {
  if (!rainfallSensorReady) return;

  rainData.rainfallTotal = RainfallSensor.getRainfall();
  rainData.rawTipCount   = RainfallSensor.getRawData();
  rainData.workingHours  = RainfallSensor.getSensorWorkingTime();

  // ── NTP-synced 1h rainfall ─────────────────────────────────────────
  // Use real wall-clock hour boundaries from NTP instead of the sensor's
  // internal window (which counts from power-on, not from top of the hour).
  struct tm timeinfo;
  if (timeInitialized && getLocalTime(&timeinfo)) {
    int currentHour = timeinfo.tm_hour;

    if (lastHourSynced == -1) {
      // First valid NTP read after boot — snapshot current tips as baseline.
      // rainfall1h will start accumulating from this point until the next
      // top-of-hour, giving a partial-hour reading that is still NTP-aligned.
      tipCountAtHourStart = rainData.rawTipCount;
      lastHourSynced      = currentHour;
      Serial.printf("[Rain] NTP hour baseline set at hour %02d, tips=%lu\n",
                    currentHour, (unsigned long)tipCountAtHourStart);
    } else if (currentHour != lastHourSynced) {
      // A new real clock-hour has started — reset the baseline.
      Serial.printf("[Rain] NTP hour boundary: %02d->%02d, resetting 1h window\n",
                    lastHourSynced, currentHour);
      tipCountAtHourStart = rainData.rawTipCount;
      lastHourSynced      = currentHour;
    }

    // Tips accumulated since the last top-of-hour × bucket size = mm
    uint32_t tipsThisHour = (rainData.rawTipCount >= tipCountAtHourStart)
                            ? (rainData.rawTipCount - tipCountAtHourStart)
                            : rainData.rawTipCount; // sensor counter wrapped/reset
    rainData.rainfall1h = tipsThisHour * RAIN_MM_PER_TIP;

  } else {
    // NTP not yet synced — fall back to sensor internal window
    rainData.rainfall1h = RainfallSensor.getRainfall(1);
  }
  // ───────────────────────────────────────────────────────────────────

  // ── NTP-synced rain rate (5-min delta window) ────────────────────────
  // Rain Rate (mm/h) = delta_Rain_Total * (60 / 5)
  // delta_Rain_Total = Rain_Total(T) - Rain_Total(T-1)
  // Slot index = floor(minute / 5) so boundaries are :00 :05 :10 ... :55
  struct tm ti2;
  if (timeInitialized && getLocalTime(&ti2)) {
    int currentSlot = (ti2.tm_min / 5);

    if (lastRateMinSlot == -1) {
      // First NTP-valid read — seed T-1 with current total
      rainTotalAtT1   = rainData.rainfallTotal;
      rainTotalAtT    = rainData.rainfallTotal;
      lastRateMinSlot = currentSlot;
      rainRate        = 0.0f;
      Serial.printf("[Rain] Rate baseline set at slot %d (%02d:%02d), total=%.2fmm\n",
                    currentSlot, ti2.tm_hour, ti2.tm_min, rainTotalAtT1);

    } else if (currentSlot != lastRateMinSlot) {
      // A new 5-min NTP slot has started:
      //   T-1 = previous T snapshot
      //   T   = current total
      //   delta = T - T-1
      rainTotalAtT1   = rainTotalAtT;             // shift: old T becomes new T-1
      rainTotalAtT    = rainData.rainfallTotal;   // new T
      float delta     = rainTotalAtT - rainTotalAtT1;
      if (delta < 0.0f) delta = 0.0f;            // guard against sensor reset
      rainRate        = delta * (60.0f / 5.0f);
      lastRateMinSlot = currentSlot;
      Serial.printf("[Rain] Rate update slot %d (%02d:%02d): T=%.2f T-1=%.2f delta=%.4f rate=%.2fmm/h\n",
                    currentSlot, ti2.tm_hour, ti2.tm_min,
                    rainTotalAtT, rainTotalAtT1, delta, rainRate);
    }
    // Between slot boundaries — keep rainRate unchanged (last computed value)
    rainData.rainRate = rainRate;

  } else {
    // NTP not synced — rate unavailable
    rainData.rainRate = 0.0f;
  }
  // ────────────────────────────────────────────────────────────────────

  rainData.valid = true;
  Serial.printf("[Rain] Total=%.2fmm 1h=%.2fmm Rate=%.2fmm/h Tips=%lu UpTime=%.2fh [%s]\n",
    rainData.rainfallTotal, rainData.rainfall1h, rainData.rainRate,
    (unsigned long)rainData.rawTipCount, rainData.workingHours,
    (timeInitialized && lastHourSynced != -1) ? "NTP-sync" : "sensor-window");
}

// ============================================================
// Time
// ============================================================
void initializeTime() {
  if (!wifiSTAConnected) return;
  configTime(gmtOffset_sec, daylightOffset_sec, ntpServer1, ntpServer2);
  time_t now = 0;
  int retry = 0;
  while (time(&now) < 1000000000 && retry < 10) { delay(500); retry++; }
  if (retry < 10) {
    timeInitialized = true;
    lastTimeSync    = millis();
    Serial.println("NTP time synchronized");
  }
}

bool getFormattedTimestamp(char* buffer, size_t bufferSize) {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) { strcpy(buffer, "0000-00-00_00:00:00"); return false; }
  strftime(buffer, bufferSize, "%d-%m-%Y_%H:%M:%S", &timeinfo);
  return true;
}

unsigned long getCurrentEpochTime() {
  time_t now; time(&now);
  return (unsigned long)now;
}

void readCalibrationOffsets() {
  float v;
  v = EEPROM.readFloat(ADDR_CAL_TEMP);     calTempOffset  = isnan(v) ? 0.0f              : v;
  v = EEPROM.readFloat(ADDR_CAL_HUMID);    calHumidOffset = isnan(v) ? 0.0f              : v;
  v = EEPROM.readFloat(ADDR_CAL_EC);       calEcOffset    = isnan(v) ? 0.0f              : v;
  v = EEPROM.readFloat(ADDR_CAL_PH);       calPhOffset    = isnan(v) ? 0.0f              : v;
  v = EEPROM.readFloat(ADDR_WATER_OFFSET); waterOffset    = isnan(v) ? 0.0f              : v;
  v = EEPROM.readFloat(ADDR_CAL_MA);       zeroMA         = (isnan(v) || v < 3.5f || v > 10.0f) ? DEFAULT_ZERO_MA : v;
  // ADDR_CAL_MA_DEPTH unused — reserved
}

void saveCalibrationOffsets() {
  EEPROM.writeFloat(ADDR_CAL_TEMP,     calTempOffset);
  EEPROM.writeFloat(ADDR_CAL_HUMID,    calHumidOffset);
  EEPROM.writeFloat(ADDR_CAL_EC,       calEcOffset);
  EEPROM.writeFloat(ADDR_CAL_PH,       calPhOffset);
  EEPROM.writeFloat(ADDR_WATER_OFFSET, waterOffset);
  EEPROM.writeFloat(ADDR_CAL_MA,       zeroMA);
  // ADDR_CAL_MA_DEPTH unused — reserved
  EEPROM.commit();
}

void readUpdateInterval() {
  unsigned long interval = 0;
  for (int i = 0; i < 4; i++)
    interval |= ((unsigned long)EEPROM.read(ADDR_UPDATE_INTERVAL + i) << (i * 8));
  if (interval == 0xFFFFFFFF || interval < 1000 || interval > 86400000) {
    serverUpdateInterval = DEFAULT_UPDATE_INTERVAL;
    for (int i = 0; i < 4; i++)
      EEPROM.write(ADDR_UPDATE_INTERVAL + i, (serverUpdateInterval >> (i * 8)) & 0xFF);
    EEPROM.commit();
  } else {
    serverUpdateInterval = interval;
  }
}

void readWiFiCredentials() {
  char s[MAX_CRED_LENGTH]={0}, p[MAX_CRED_LENGTH]={0};
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    s[i] = EEPROM.read(ADDR_WIFI_SSID + i);
    p[i] = EEPROM.read(ADDR_WIFI_PASS + i);
  }

  sta_password = String(p); sta_password.trim();
}

void readServerConfig() {
  char h[MAX_SERVER_LENGTH]={0}, p[MAX_SERVER_LENGTH]={0};
  for (int i = 0; i < MAX_SERVER_LENGTH; i++) {
    h[i] = EEPROM.read(ADDR_SERVER_HOST + i);
    p[i] = EEPROM.read(ADDR_SERVER_PATH + i);
  }
  serverHost = String(h); serverHost.trim();
  serverPath = String(p); serverPath.trim();
  serverPort = EEPROM.read(ADDR_SERVER_PORT) | (EEPROM.read(ADDR_SERVER_PORT + 1) << 8);
  readUpdateInterval();
  bool ok = !serverHost.isEmpty() && serverHost.indexOf('.') != -1
         && !serverPath.isEmpty() && serverPath.charAt(0) == '/'
         && serverPort > 0 && serverPort <= 65535;
  if (!ok) lastServerStatus = "Not configured";
}

void saveWiFiCredentials(String ssid, String pass, String appass) {
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_SSID + i, 0);
    EEPROM.write(ADDR_WIFI_PASS + i, 0);
  }
  for (int i = 0; i < (int)ssid.length() && i < MAX_CRED_LENGTH; i++)
    EEPROM.write(ADDR_WIFI_SSID + i, ssid[i]);
  for (int i = 0; i < (int)pass.length() && i < MAX_CRED_LENGTH; i++)
    EEPROM.write(ADDR_WIFI_PASS + i, pass[i]);
  EEPROM.commit();
  sta_ssid = ssid; sta_password = pass;
}

void saveUpdateInterval(unsigned long interval) {
  for (int i = 0; i < 4; i++)
    EEPROM.write(ADDR_UPDATE_INTERVAL + i, (interval >> (i * 8)) & 0xFF);
  serverUpdateInterval = interval;
}

void saveServerConfig(String host, String path, int port, unsigned long interval) {
  for (int i = 0; i < MAX_SERVER_LENGTH; i++) {
    EEPROM.write(ADDR_SERVER_HOST + i, 0);
    EEPROM.write(ADDR_SERVER_PATH + i, 0);
  }
  for (int i = 0; i < (int)host.length() && i < MAX_SERVER_LENGTH; i++)
    EEPROM.write(ADDR_SERVER_HOST + i, host[i]);
  for (int i = 0; i < (int)path.length() && i < MAX_SERVER_LENGTH; i++)
    EEPROM.write(ADDR_SERVER_PATH + i, path[i]);
  EEPROM.write(ADDR_SERVER_PORT,     port & 0xFF);
  EEPROM.write(ADDR_SERVER_PORT + 1, (port >> 8) & 0xFF);
  saveUpdateInterval(interval);
  EEPROM.commit();
  serverHost = host; serverPath = path; serverPort = port; serverUpdateInterval = interval;
}

void initializeWiFi() {
  Serial.println("Initializing WiFi...");
  WiFi.mode(WIFI_AP_STA);
  // AP starts only if button is held at boot; otherwise AP is off
  // (AP state is managed by checkAPButton() in loop)

  if (sta_ssid.length() > 0) {
    WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    WiFi.setTxPower(WIFI_POWER_8_5dBm);
    unsigned long t = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - t < 10000) { delay(500); Serial.print("."); }
    if (WiFi.status() == WL_CONNECTED) {
      wifiSTAConnected = true;
      Serial.println("\nWiFi OK — IP: " + WiFi.localIP().toString());
      initializeTime();
    } else {
      Serial.println("\nWiFi STA failed");
    }
  }
}

// ============================================================
// AP Button (PIN 36, internal pull-up, active-LOW)
// Pressed (LOW)  → AP ON   |   Released (HIGH) → AP OFF
// ============================================================
void checkAPButton() {
  bool buttonPressed = (digitalRead(AP_BUTTON_PIN) == LOW);

  if (buttonPressed && !apEnabled) {
    // Button just engaged → turn AP ON (open network, no password)
    WiFi.softAP(ap_ssid);
    apEnabled = true;
    Serial.println("[AP] Button engaged — TMAT-Config AP started (open)");
  } else if (!buttonPressed && apEnabled) {
    // Button released → turn AP OFF
    WiFi.softAPdisconnect(true);
    apEnabled = false;
    Serial.println("[AP] Button disengaged — TMAT-Config AP stopped");
  }
}

void checkWiFiConnection() {
  if (millis() - lastWifiReconnect < 30000) return;
  lastWifiReconnect = millis();
  if (WiFi.status() != WL_CONNECTED) {
    wifiSTAConnected = false;
    WiFi.reconnect();
    unsigned long t = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - t < 10000) { delay(500); Serial.print("."); }
    if (WiFi.status() == WL_CONNECTED) {
      wifiSTAConnected = true;
      Serial.println("\nWiFi reconnected!");
      if (!timeInitialized) initializeTime();
    }
  }
}

String getSDFilePath(time_t epoch) {
  struct tm t; localtime_r(&epoch, &t);
  char buf[32]; strftime(buf, sizeof(buf), "/data/%Y-%m-%d.txt", &t);
  return String(buf);
}

const char* SD_HEADER = "epoch,timestamp,water_depth,sensor_raw,current_ma,"
                        "soil_temp,soil_humid,soil_ec,soil_ph,"
                        "rain_total,rain_1h,rain_rate,rain_tips,status";

String buildCSVLine(float waterDepth, const char* status) {
  getFormattedTimestamp(timeStampBuf, sizeof(timeStampBuf));
  String line = String(getCurrentEpochTime()) + "," + String(timeStampBuf) + ","
              + String(waterDepth, 2) + "," + String(distance_cm, 2) + ","
              + String(currentMA, 3)  + ",";
  if (soilData.valid) {
    line += String(round(soilData.temperature * 10)/10.0, 1) + ","
          + String(round(soilData.humidity    * 10)/10.0, 1) + ","
          + String((int)soilData.conductivity)                + ","
          + String(round(soilData.ph          * 10)/10.0, 1) + ",";
  } else { line += ",,,,"; }
  if (rainData.valid) {
    line += String(round(rainData.rainfallTotal*100)/100.0, 2) + ","
          + String(round(rainData.rainfall1h   *100)/100.0, 2) + ","
          + String(round(rainData.rainRate      *100)/100.0, 2) + ","
          + String((int)rainData.rawTipCount) + ",";
  } else { line += ",,,,"; }
  line += String(status);
  return line;
}

void writeSDRecord(float waterDepth, const char* status) {
  if (!sdAvailable) return;
  time_t epoch = getCurrentEpochTime();
  if (epoch < 1000000) return;
  String path = getSDFilePath(epoch);
  if (!SD.exists(path)) {
    File f = SD.open(path, FILE_WRITE);
    if (f) { f.println(SD_HEADER); f.close(); }
  }
  File f = SD.open(path, FILE_APPEND);
  if (f) { f.println(buildCSVLine(waterDepth, status)); f.close();
    Serial.printf("[SD] Written %s [%s]\n", path.c_str(), status);
  }
}

time_t nextAligned5Min(time_t now) { return ((now / 300) + 1) * 300; }

void checkSDSchedule(float waterDepth) {
  if (!sdAvailable) return;
  time_t now = getCurrentEpochTime();
  if (now < 1000000) return;
  if (nextSDRecord == 0) {
    nextSDRecord = nextAligned5Min(now);
    Serial.printf("[SD] First record at epoch %lu\n", (unsigned long)nextSDRecord);
    return;
  }
  if (now >= nextSDRecord) {
    writeSDRecord(waterDepth, "LIVE");
    nextSDRecord += 300;
    Serial.printf("[SD] Next record at epoch %lu\n", (unsigned long)nextSDRecord);
  }
}

bool replaySendLine(const String& line) {
  if (line.startsWith("epoch") || line.length() < 10) return false;
  int idx = 0; String fields[14]; int fc = 0;
  while (idx < (int)line.length() && fc < 14) {
    int c = line.indexOf(',', idx);
    if (c == -1) c = line.length();
    fields[fc++] = line.substring(idx, c); idx = c + 1;
  }
  if (fc < 5) return false;
  JsonDocument doc;
  doc["device_id"]   = DEVICE_ID;
  doc["epoch_time"]  = (unsigned long)fields[0].toInt();
  doc["timestamp"]   = fields[1];
  doc["water_depth"] = fields[2].toFloat();
  doc["sensor_raw"]  = fields[3].toFloat();
  doc["current_ma"]  = fields[4].toFloat();
  if (fields[5].length()  > 0) doc["soil_temp"]  = fields[5].toFloat();
  if (fields[6].length()  > 0) doc["soil_humid"] = fields[6].toFloat();
  if (fields[7].length()  > 0) doc["soil_ec"]    = fields[7].toInt();
  if (fields[8].length()  > 0) doc["soil_ph"]    = fields[8].toFloat();
  if (fields[9].length()  > 0) doc["rain_total"] = fields[9].toFloat();
  if (fields[10].length() > 0) doc["rain_1h"]    = fields[10].toFloat();
  if (fields[11].length() > 0) doc["rain_rate"]  = fields[11].toFloat();
  if (fields[12].length() > 0) doc["rain_tips"]  = fields[12].toInt();
  doc["recovered"] = true;
  String json; serializeJson(doc, json);
  WiFiClientSecure client; client.setInsecure();
  if (!client.connect(serverHost.c_str(), serverPort)) return false;
  client.print("POST " + serverPath + " HTTP/1.1\r\nHost: " + serverHost
    + "\r\nContent-Type: application/json\r\nContent-Length: "
    + json.length() + "\r\nConnection: close\r\n\r\n" + json);
  unsigned long t = millis(); int code = 0;
  while (client.connected() && millis()-t < 8000) {
    while (client.available()) {
      String ln = client.readStringUntil('\n');
      if (ln.startsWith("HTTP/1.1")) {
        int s1=ln.indexOf(' '), s2=ln.indexOf(' ',s1+1);
        if (s1>0) code=ln.substring(s1+1,s2).toInt();
      }
    }
  }
  client.stop();
  Serial.printf("[SD Replay] epoch=%s HTTP=%d\n", fields[0].c_str(), code);
  return (code == 200);
}


// ---- Replay queue: up to 8 pending files, oldest first ----
#define REPLAY_MAX_FILES 8
String replayQueue[REPLAY_MAX_FILES];
int    replayQueueHead  = 0;
int    replayQueueCount = 0;

// Advance replay to the next queued file (called when current file finishes)
void advanceReplayQueue() {
  replayQueueHead++;
  if (replayQueueHead >= replayQueueCount) {
    sdReplayActive  = false;
    replayQueueHead = 0;
    replayQueueCount = 0;
    Serial.println("[SD Replay] All files replayed.");
    return;
  }
  replayFile      = replayQueue[replayQueueHead];
  replayLineIndex = 0;
  Serial.printf("[SD Replay] Advancing to next file: %s\n", replayFile.c_str());
}

void replayNextRecord() {
  if (!sdReplayActive || !sdAvailable || !wifiSTAConnected) return;
  if (serverHost.isEmpty() || serverPath.isEmpty() || serverPort <= 0) return;
  File f = SD.open(replayFile, FILE_READ);
  if (!f) { advanceReplayQueue(); return; }
  int lineNum = 0; String line = "";
  while (f.available()) {
    line = f.readStringUntil('\n'); line.trim();
    if (lineNum == replayLineIndex) break;
    lineNum++;
  }
  f.close();
  if (lineNum < replayLineIndex || line.length() < 10) {
    // Current file exhausted — move to next file in queue
    Serial.printf("[SD Replay] Complete — %s done\n", replayFile.c_str());
    advanceReplayQueue(); return;
  }
  if (!line.startsWith("epoch")) replaySendLine(line);
  replayLineIndex++;
}

void startReplay() {
  if (!sdAvailable) return;
  time_t fromEpoch = loadLastSendEpoch();
  if (fromEpoch == 0) return;
  time_t now = getCurrentEpochTime();

  // Build queue oldest-first (d=7 is oldest day, d=0 is today)
  replayQueueHead  = 0;
  replayQueueCount = 0;
  for (int d = 7; d >= 0; d--) {
    String path = getSDFilePath(now - d * 86400);
    if (!SD.exists(path)) continue;
    // Check if this file has any record newer than fromEpoch
    File f = SD.open(path, FILE_READ); if (!f) continue;
    bool hasNew = false;
    while (f.available()) {
      String line = f.readStringUntil('\n'); line.trim();
      if (line.startsWith("epoch") || line.length() < 5) continue;
      int comma = line.indexOf(',');
      if (comma > 0 && (time_t)line.substring(0, comma).toInt() > fromEpoch) {
        hasNew = true; break;
      }
    }
    f.close();
    if (hasNew && replayQueueCount < REPLAY_MAX_FILES) {
      replayQueue[replayQueueCount++] = path;
      Serial.printf("[SD Replay] Queued: %s\n", path.c_str());
    }
  }

  if (replayQueueCount == 0) {
    Serial.println("[SD Replay] No missed records");
    return;
  }

  // Start with the first (oldest) file, seek past already-sent lines
  replayFile      = replayQueue[0];
  replayLineIndex = 0;
  sdReplayActive  = true;

  File f = SD.open(replayFile, FILE_READ); if (!f) return;
  int ln = 0;
  while (f.available()) {
    String line = f.readStringUntil('\n'); line.trim(); ln++;
    if (line.startsWith("epoch") || line.length() < 5) continue;
    int comma = line.indexOf(',');
    if (comma > 0 && (time_t)line.substring(0, comma).toInt() > fromEpoch) {
      replayLineIndex = ln - 1;
      Serial.printf("[SD Replay] Start %s from line %d\n", replayFile.c_str(), replayLineIndex);
      break;
    }
  }
  f.close();
}

void setupSD() {
  SPI.begin(SD_SCK_PIN, SD_MISO_PIN, SD_MOSI_PIN, SD_CS_PIN);
  if (!SD.begin(SD_CS_PIN)) {
    Serial.println("[SD] Not found — logging disabled");
    sdAvailable = false; return;
  }
  sdAvailable = true;
  if (!SD.exists("/data"))  SD.mkdir("/data");
  if (!SD.exists("/state")) SD.mkdir("/state");
  lastSuccessfulSendEpoch = loadLastSendEpoch();
  uint64_t mb = SD.cardSize()/(1024*1024);
  Serial.printf("[SD] Ready %lluMB  lastSend=%lu\n", mb, (unsigned long)lastSuccessfulSendEpoch);
}

// ============================================================
bool sendToServer(float waterDepth) {
  if (!wifiSTAConnected || serverHost.isEmpty() || serverPath.isEmpty() || serverPort <= 0)
    return false;
  if (millis() - lastServerAttempt < serverUpdateInterval) return false;
  lastServerAttempt = millis();

  Serial.println("--- Sending to server ---");
  WiFiClientSecure client;
  client.setInsecure();
  if (!client.connect(serverHost.c_str(), serverPort)) {
    lastServerStatus = "Connection failed";
    Serial.println("Server connect FAILED");
    return false;
  }

  getFormattedTimestamp(timeStampBuf, sizeof(timeStampBuf));

  JsonDocument doc;
  doc["device_id"]     = DEVICE_ID;
  doc["epoch_time"]    = getCurrentEpochTime();
  doc["timestamp"]     = timeStampBuf;
  // Water level
  doc["water_depth"]   = waterDepth;
  doc["sensor_raw"]    = distance_cm;
  doc["current_ma"]    = currentMA;
  doc["zero_ma"]       = zeroMA;
  // Soil
  if (soilData.valid) {
    doc["soil_temp"]  = round(soilData.temperature  * 10) / 10.0;
    doc["soil_humid"] = round(soilData.humidity     * 10) / 10.0;
    doc["soil_ec"]    = (int)soilData.conductivity;
    doc["soil_ph"]    = round(soilData.ph           * 10) / 10.0;
  }
  // Rainfall
  if (rainData.valid) {
    doc["rain_total"] = round(rainData.rainfallTotal * 100) / 100.0;
    doc["rain_1h"]    = round(rainData.rainfall1h    * 100) / 100.0;
    doc["rain_rate"]  = round(rainData.rainRate      * 100) / 100.0; // mm/h (5-min NTP delta)
    doc["rain_tips"]  = rainData.rawTipCount;
  }

  String json;
  serializeJson(doc, json);
  Serial.println("JSON: " + json);

  client.print("POST " + serverPath + " HTTP/1.1\r\n"
             + "Host: " + serverHost + "\r\n"
             + "Content-Type: application/json\r\n"
             + "Content-Length: " + json.length() + "\r\n"
             + "Connection: close\r\n\r\n"
             + json);

  unsigned long t = millis();
  String response = "";
  int code = 0;
  while (client.connected() && millis() - t < 10000) {
    while (client.available()) {
      String line = client.readStringUntil('\n');
      response += line + "\n";
      if (line.startsWith("HTTP/1.1")) {
        int s1 = line.indexOf(' '), s2 = line.indexOf(' ', s1+1);
        if (s1 > 0) code = line.substring(s1+1, s2).toInt();
      }
    }
  }
  client.stop();

  if (code == 200 || response.indexOf("success") >= 0) {
    lastServerSend          = millis();
    lastServerStatus        = "OK (HTTP " + String(code) + ")";
    serverConnected         = true;
    time_t ep               = getCurrentEpochTime();
    lastSuccessfulSendEpoch = ep;
    saveLastSendEpoch(ep);
  } else {
    lastServerStatus = "Error HTTP " + String(code);
    serverConnected  = false;
    writeSDRecord(waterDepth, "OFFLINE");
  }
  Serial.println("Server: " + lastServerStatus);
  return serverConnected;
}

void handleOTAUpload() {
  HTTPUpload& upload = server.upload();

  if (upload.status == UPLOAD_FILE_START) {
    // Freeze all measurements so OTA gets full CPU + bus access
    otaInProgress = true;
    Serial.printf("\n[OTA] START: %s - suspending all measurements\n",
                  upload.filename.c_str());
    // Disconnect STA background tasks; AP stays up for browser
    WiFi.disconnect(false);
    delay(100);
    if (!Update.begin(UPDATE_SIZE_UNKNOWN)) {
      Update.printError(Serial);
      otaInProgress = false;
    }

  } else if (upload.status == UPLOAD_FILE_WRITE) {
    if (Update.write(upload.buf, upload.currentSize) != upload.currentSize) {
      Update.printError(Serial);
    } else {
      static uint32_t lastPrint = 0;
      if (millis() - lastPrint > 1000) {
        lastPrint = millis();
        Serial.printf("[OTA] Written %u bytes...\n", upload.totalSize);
      }
    }

  } else if (upload.status == UPLOAD_FILE_END) {
    if (Update.end(true)) {
      Serial.printf("[OTA] SUCCESS: %u bytes - rebooting\n", upload.totalSize);
      server.send(200, "text/plain", "Update successful! Rebooting...");
      delay(500);
      ESP.restart();
    } else {
      Update.printError(Serial);
      server.send(500, "text/plain", "Update FAILED - device NOT rebooted. Try again.");
      otaInProgress = false;
      Serial.println("[OTA] FAILED - resuming normal operation");
      WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    }

  } else if (upload.status == UPLOAD_FILE_ABORTED) {
    Update.abort();
    otaInProgress = false;
    Serial.println("[OTA] ABORTED - resuming normal operation");
    WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
  }
}

  Serial.printf("Cal loaded: zeroMA=%.2fmA offset=%.1fcm T%+.2f H%+.2f EC%+.1f pH%+.2f\n",
    zeroMA, waterOffset, calTempOffset, calHumidOffset, calEcOffset, calPhOffset);

  // ADC: 4-20mA submersible water level sensor on GPIO4
  analogReadResolution(12);
  analogSetAttenuation(ADC_11db);
  pinMode(WATER_LEVEL_ADC_PIN, INPUT);
  Serial.printf("ADC Water Level: GPIO%d  shunt=%.0fΩ  zeroMA=%.2fmA  offset=%.1fcm\n",
    WATER_LEVEL_ADC_PIN, SHUNT_OHMS, zeroMA, waterOffset);

  // UART2: Soil sensor via RS485-TTL
  SoilSerial.begin(4800, SERIAL_8N1, SOIL_RX_PIN, SOIL_TX_PIN);
  Serial.printf("UART2 Soil RS485: RX=%d TX=%d @ 4800\n", SOIL_RX_PIN, SOIL_TX_PIN);

  // I2C: DFRobot Rainfall sensor (SDA=8, SCL=9 on ESP32-S3)
  Wire.begin();
  setupSD();
  Serial.println("I2C started — initializing DFRobot SEN0575 rainfall sensor...");
  if (RainfallSensor.begin()) {
    rainfallSensorReady = true;
    Serial.println("Rainfall sensor OK!");
    Serial.printf("  VID: 0x%04X  PID: 0x%04X\n", RainfallSensor.vid, RainfallSensor.pid);
    Serial.println("  FW: " + RainfallSensor.getFirmwareVersion());
  } else {
    Serial.println("Rainfall sensor FAILED — check I2C wiring (SDA=8, SCL=9)");
  }

  initializeWiFi();
  // Check button state at boot so AP starts immediately if held
  checkAPButton();
  WiFi.scanNetworks(true);

  // Routes
  server.on("/",            HTTP_GET,  handleRoot);
  server.on("/readings",    HTTP_GET,  []() { server.send(200, "text/plain", generateReadingsHtml()); });
  server.on("/wifi-config", HTTP_GET,  handleWiFiConfig);
  server.on("/save-sta",    HTTP_POST, handleSaveSTA);
  server.on("/save-ap",     HTTP_POST, handleSaveAP);
  server.on("/save-server", HTTP_POST, handleSaveServer);
  server.on("/sen-cal",     HTTP_GET,  handleSensorCalibration);
  server.on("/save-cal",    HTTP_POST, handleSaveCalibration);
  server.on("/ota-update",  HTTP_GET,  handleOTAUpdate);
  server.on("/ota-upload",  HTTP_POST, []() { server.send(200); }, handleOTAUpload);
  server.begin();

  Serial.println("\nWeb server started:");
  Serial.println("  192.168.4.1/           Main dashboard");
  Serial.println("  192.168.4.1/sen-cal    Sensor calibration");
  Serial.println("  192.168.4.1/wifi-config  WiFi & server settings");
  Serial.println("  192.168.4.1/ota-update   Firmware update");
  Serial.println("========================================");
}

// ============================================================
// loop()
// ============================================================
void loop() {
  checkAPButton();
  server.handleClient();

  if (otaInProgress) { delay(10); return; }

  checkWiFiConnection();

  // Detect WiFi reconnection → trigger replay of missed SD records
  if (wifiSTAConnected && !wifiWasConnected) {
    Serial.println("[SD] WiFi reconnected — checking for missed records");
    startReplay();
  }
  wifiWasConnected = wifiSTAConnected;

  static unsigned long lastSensorRead  = 0;
  static unsigned long lastStatusPrint = 0;

  if (millis() - lastSensorRead >= 2000) {
    lastSensorRead = millis();
    readSoilSensor();
    readRainfallSensor();
    WaterLevelData wd = readWaterLevel();
    // TEST MODE: always use waterDepth directly, skip isValid check
    float wDepth = wd.waterDepth;

    // SD: NTP-aligned 5-min record (always, WiFi independent)
    checkSDSchedule(wDepth);

    // SD: replay one missed record per cycle (non-blocking)
    replayNextRecord();

    // Server: send live data on interval (no valid check — testing)
    if (wifiSTAConnected && !serverHost.isEmpty() && !serverPath.isEmpty() && serverPort > 0) {
      serverConnected = sendToServer(wDepth);
    }
  }

  if (millis() - lastStatusPrint >= 10000) {
    lastStatusPrint = millis();
    getFormattedTimestamp(timeStampBuf, sizeof(timeStampBuf));
    Serial.println("\n=== Status @ " + String(timeStampBuf) + " ===");
    Serial.printf("Water  : %.2fmA  pressureDepth=%.1fcm  waterDepth=%.1fcm (GPIO%d)\n",
      currentMA, distance_cm, distance_cm - waterOffset, WATER_LEVEL_ADC_PIN);
    Serial.printf("Soil   : T=%.1f H=%.1f EC=%.0f pH=%.1f [%s]\n",
      soilData.temperature, soilData.humidity, soilData.conductivity, soilData.ph,
      soilData.valid ? "OK" : "ERR");
    Serial.printf("Rain   : %.2fmm total | %.2fmm/1h | %lu tips [%s]\n",
      rainData.rainfallTotal, rainData.rainfall1h, (unsigned long)rainData.rawTipCount,
      rainData.valid ? "OK" : "ERR");
    Serial.printf("WiFi   : %s | RSSI:%ddBm | AP:%s clients:%d\n",
      wifiSTAConnected ? WiFi.localIP().toString().c_str() : "Disconnected",
      WiFi.RSSI(), apEnabled ? "ON" : "OFF", WiFi.softAPgetStationNum());
    Serial.printf("Server : %s\n", lastServerStatus.c_str());
    Serial.printf("SD     : %s | Replay:%s | NextRecord in %lus\n",
      sdAvailable ? "OK" : "NO CARD",
      sdReplayActive ? replayFile.c_str() : "idle",
      nextSDRecord > 0 ? (unsigned long)(nextSDRecord - getCurrentEpochTime()) : 0UL);
    Serial.println("=================================");
  }

  delay(10);
}
