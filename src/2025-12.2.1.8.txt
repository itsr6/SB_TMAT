// Time functions
void initializeTime() {
  if (wifiSTAConnected) {
    configTime(gmtOffset_sec, daylightOffset_sec, ntpServer1, ntpServer2);
    time_t now = 0;
    int retry = 0;
    while (time(&now) < 1000000000 && retry < 10) {
      delay(500);
      retry++;
    }
    if (retry < 10) {
      timeInitialized = true;
      Serial.println("Time initialized successfully!");
    }
  }
}

bool getFormattedTimestamp(char* buffer, size_t bufferSize) {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    strcpy(buffer, "0000-00-00_00:00:00");
    return false;
  }
  strftime(buffer, bufferSize, "%d-%m-%Y_%H:%M:%S", &timeinfo);
  return true;
}

unsigned long getCurrentEpochTime() {
  time_t now;
  time(&now);
  return (unsigned long)now;
}

// EEPROM functions
void readWiFiCredentials() {
  char ssid_buf[MAX_CRED_LENGTH] = {0};
  char pass_buf[MAX_CRED_LENGTH] = {0};
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    ssid_buf[i] = EEPROM.read(ADDR_WIFI_SSID + i);
    pass_buf[i] = EEPROM.read(ADDR_WIFI_PASS + i);
  }
  sta_ssid = String(ssid_buf); sta_ssid.trim();
  sta_password = String(pass_buf); sta_password.trim();
}

void readAPPassword() {
  char ap_pass_buf[MAX_CRED_LENGTH] = {0};
  bool valid = false;
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    ap_pass_buf[i] = EEPROM.read(ADDR_AP_PASS + i);
    if (ap_pass_buf[i] != 0 && ap_pass_buf[i] != 255) valid = true;
  }
  ap_password = String(ap_pass_buf); ap_password.trim();
  if (!valid || ap_password.isEmpty()) {
    ap_password = "12345678";
    for (int i = 0; i < (int)ap_password.length(); i++)
      EEPROM.write(ADDR_AP_PASS + i, ap_password[i]);
    EEPROM.commit();
  }
}

void readUpdateInterval() {
  unsigned long interval = 0;
  for (int i = 0; i < 4; i++)
    interval |= ((unsigned long)EEPROM.read(ADDR_UPDATE_INTERVAL + i) << (i * 8));
  if (interval == 0xFFFFFFFF || interval < 1000 || interval > 86400000) {
    serverUpdateInterval = DEFAULT_UPDATE_INTERVAL;
    saveUpdateInterval(serverUpdateInterval);
  } else {
    serverUpdateInterval = interval;
  }
  Serial.printf("Server update interval: %lu seconds\n", serverUpdateInterval / 1000);
}

void readServerConfig() {
  char host_buf[MAX_SERVER_LENGTH] = {0};
  char path_buf[MAX_SERVER_LENGTH] = {0};
  for (int i = 0; i < MAX_SERVER_LENGTH; i++) {
    host_buf[i] = EEPROM.read(ADDR_SERVER_HOST + i);
    path_buf[i] = EEPROM.read(ADDR_SERVER_PATH + i);
  }
  serverHost = String(host_buf); serverHost.trim();
  serverPath = String(path_buf); serverPath.trim();
  serverPort = EEPROM.read(ADDR_SERVER_PORT) | (EEPROM.read(ADDR_SERVER_PORT + 1) << 8);
  readUpdateInterval();

  bool configValid = true;
  if (serverHost.isEmpty() || serverHost.indexOf('.') == -1) { configValid = false; Serial.println("Server host not configured or invalid"); }
  if (serverPath.isEmpty() || serverPath.charAt(0) != '/')   { configValid = false; Serial.println("Server path not configured or invalid"); }
  if (serverPort <= 0 || serverPort > 65535)                  { configValid = false; Serial.println("Server port not configured or invalid"); }

  if (!configValid) {
    lastServerStatus = "Not configured - Please set up server in web interface";
  } else {
    Serial.println("Server configuration loaded from EEPROM");
  }
}

void saveUpdateInterval(unsigned long interval) {
  for (int i = 0; i < 4; i++)
    EEPROM.write(ADDR_UPDATE_INTERVAL + i, (interval >> (i * 8)) & 0xFF);
  serverUpdateInterval = interval;
}

void saveWiFiCredentials(String ssid, String password, String ap_pass) {
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_SSID + i, 0);
    EEPROM.write(ADDR_WIFI_PASS + i, 0);
    if (!ap_pass.isEmpty()) EEPROM.write(ADDR_AP_PASS + i, 0);
  }
  for (int i = 0; i < (int)ssid.length() && i < MAX_CRED_LENGTH; i++)
    EEPROM.write(ADDR_WIFI_SSID + i, ssid[i]);
  for (int i = 0; i < (int)password.length() && i < MAX_CRED_LENGTH; i++)
    EEPROM.write(ADDR_WIFI_PASS + i, password[i]);
  if (!ap_pass.isEmpty())
    for (int i = 0; i < (int)ap_pass.length() && i < MAX_CRED_LENGTH; i++)
      EEPROM.write(ADDR_AP_PASS + i, ap_pass[i]);
  EEPROM.commit();
  sta_ssid = ssid;
  sta_password = password;
}

void saveServerConfig(String host, String path, int port, unsigned long interval) {
  for (int i = 0; i < MAX_SERVER_LENGTH; i++) {
    EEPROM.write(ADDR_SERVER_HOST + i, 0);
    EEPROM.write(ADDR_SERVER_PATH + i, 0);
  }
  for (int i = 0; i < (int)host.length() && i < MAX_SERVER_LENGTH; i++)
    EEPROM.write(ADDR_SERVER_HOST + i, host[i]);
  for (int i = 0; i < (int)path.length() && i < MAX_SERVER_LENGTH; i++)
    EEPROM.write(ADDR_SERVER_PATH + i, path[i]);
  EEPROM.write(ADDR_SERVER_PORT,     port & 0xFF);
  EEPROM.write(ADDR_SERVER_PORT + 1, (port >> 8) & 0xFF);
  saveUpdateInterval(interval);
  EEPROM.commit();
  serverHost = host;
  serverPath = path;
  serverPort = port;
  serverUpdateInterval = interval;
  Serial.printf("Server config saved -> Host:%s Path:%s Port:%d Interval:%lus\n",
                host.c_str(), path.c_str(), port, interval / 1000);
}

// Soil Moisture
void readSoilCalibration() {
  const int dryAddrs[NUM_SOIL_SENSORS] = {ADDR_SOIL1_DRY, ADDR_SOIL2_DRY, ADDR_SOIL3_DRY, ADDR_SOIL4_DRY};
  const int wetAddrs[NUM_SOIL_SENSORS] = {ADDR_SOIL1_WET, ADDR_SOIL2_WET, ADDR_SOIL3_WET, ADDR_SOIL4_WET};
  bool allValid = true;
  for (int i = 0; i < NUM_SOIL_SENSORS; i++) {
    int dry = 0, wet = 0;
    for (int b = 0; b < 4; b++) {
      dry |= (EEPROM.read(dryAddrs[i] + b) << (b * 8));
      wet |= (EEPROM.read(wetAddrs[i] + b) << (b * 8));
    }
    if (dry >= 0 && dry <= 4095 && wet >= 0 && wet <= 4095 && dry != wet) {
      soilDryValues[i] = dry;
      soilWetValues[i] = wet;
      soilCalibrated[i] = true;
    } else {
      soilDryValues[i] = DEFAULT_SOIL_DRY;
      soilWetValues[i] = DEFAULT_SOIL_WET;
      soilCalibrated[i] = false;
      allValid = false;
    }
  }
  if (!allValid) {
    for (int i = 0; i < NUM_SOIL_SENSORS; i++)
      saveSoilCalibration(i, soilDryValues[i], soilWetValues[i]);
    Serial.println("Soil calibration defaults saved.");
  }
  for (int i = 0; i < NUM_SOIL_SENSORS; i++)
    Serial.printf("  Soil %d: dry=%d wet=%d\n", i+1, soilDryValues[i], soilWetValues[i]);
}

void saveSoilCalibration(int idx, int dry, int wet) {
  if (idx < 0 || idx >= NUM_SOIL_SENSORS) return;
  const int dryAddrs[NUM_SOIL_SENSORS] = {ADDR_SOIL1_DRY, ADDR_SOIL2_DRY, ADDR_SOIL3_DRY, ADDR_SOIL4_DRY};
  const int wetAddrs[NUM_SOIL_SENSORS] = {ADDR_SOIL1_WET, ADDR_SOIL2_WET, ADDR_SOIL3_WET, ADDR_SOIL4_WET};
  for (int i = 0; i < 4; i++) {
    EEPROM.write(dryAddrs[idx] + i, (dry >> (i * 8)) & 0xFF);
    EEPROM.write(wetAddrs[idx] + i, (wet >> (i * 8)) & 0xFF);
  }
  EEPROM.commit();
  soilDryValues[idx] = dry;
  soilWetValues[idx] = wet;
  soilCalibrated[idx] = true;
  Serial.printf("Soil calib saved sensor %d: dry=%d wet=%d\n", idx+1, dry, wet);
}

void readSoilMoisture() {
  for (int i = 0; i < NUM_SOIL_SENSORS; i++) {
    soilRaw[i]     = analogRead(soilPins[i]);
    soilPercent[i] = computeSoilPercent(soilRaw[i], soilDryValues[i], soilWetValues[i]);
  }
}

int computeSoilPercent(int raw, int dry, int wet) {
  if (dry == wet) return 0;
  if (raw >= wet) return 100;
  if (raw <= dry) return 0;
  return map(raw, dry, wet, 0, 100);
}

// WiFi
void initializeWiFi() {
  Serial.println("Initializing WiFi...");
  WiFi.mode(WIFI_AP_STA);
  if (!WiFi.softAP(ap_ssid, ap_password.c_str()))
    WiFi.softAP(ap_ssid, "12345678");

  if (sta_ssid.length() > 0) {
    WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    WiFi.setTxPower(WIFI_POWER_8_5dBm);
    unsigned long startTime = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - startTime < 10000) {
      delay(500); Serial.print(".");
    }
    if (WiFi.status() == WL_CONNECTED) {
      wifiSTAConnected = true;
      Serial.println("\nWiFi connected! IP: " + WiFi.localIP().toString());
      initializeTime();
    } else {
      Serial.println("\nWiFi connection failed.");
    }
  }
}

void checkWiFiConnection() {
  static unsigned long lastCheck = 0;
  if (millis() - lastCheck < 30000) return;
  lastCheck = millis();
  if (WiFi.status() != WL_CONNECTED) {
    wifiSTAConnected = false;
    Serial.println("WiFi disconnected, reconnecting...");
    WiFi.reconnect();
    unsigned long startTime = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - startTime < 10000) {
      delay(500); Serial.print(".");
    }
    if (WiFi.status() == WL_CONNECTED) {
      wifiSTAConnected = true;
      Serial.println("\nWiFi reconnected!");
      if (!timeInitialized) initializeTime();
    } else {
      Serial.println("\nWiFi reconnection failed.");
    }
  }
}

// sendToServer — FIXED: uses HTTPClient, correct Host header, HTTP/HTTPS auto-detect
bool sendToServer(float waterDepth) {
  if (!wifiSTAConnected) {
    lastServerStatus = "WiFi not connected";
    return false;
  }
  if (serverHost.isEmpty() || serverPath.isEmpty() || serverPort <= 0) {
    lastServerStatus = "Server not configured";
    return false;
  }
  if (millis() - lastServerAttempt < serverUpdateInterval) return false;

  lastServerAttempt = millis();

  getFormattedTimestamp(timeStampBuf, sizeof(timeStampBuf));
  distance_cm = distance / 10.0;

  DynamicJsonDocument doc(1024);
  doc["device_id"]     = DEVICE_ID;
  doc["sensor_raw"]    = distance_cm;
  doc["water_depth"]   = waterDepth;
  doc["sensor_height"] = sensorHeight;
  doc["epoch_time"]    = getCurrentEpochTime();
  doc["soil_sensor1"]  = soilPercent[0];
  doc["soil_sensor2"]  = soilPercent[1];
  doc["soil_sensor3"]  = soilPercent[2];
  doc["soil_sensor4"]  = soilPercent[3];

  String jsonString;
  serializeJson(doc, jsonString);

  Serial.println("========================================");
  Serial.println("Sending data to server...");
  Serial.println(jsonString);

  // ✅ FIX: Declare clients at function scope so they are NOT destroyed before http.POST() runs.
  //         The "Bad file number" errno:9 error was caused by WiFiClient going out of scope
  //         inside the if-block while HTTPClient still held a reference to it.
  bool useSSL = (serverPort == 443);
  String url  = (useSSL ? "https://" : "http://") + serverHost + ":" + String(serverPort) + serverPath;
  Serial.println("URL: " + url);

  WiFiClient        plainClient;
  WiFiClientSecure  secureClient;
  HTTPClient        http;
  bool begun = false;

  if (useSSL) {
    secureClient.setInsecure();  // Accept self-signed certs
    begun = http.begin(secureClient, url);
  } else {
    begun = http.begin(plainClient, url);
  }

  if (!begun) {
    Serial.println("HTTPClient.begin() failed");
    lastServerStatus = "Connection failed";
    serverConnected  = false;
    http.end();
    return false;
  }

  http.addHeader("Content-Type", "application/json");
  http.setTimeout(10000);

  int httpCode = http.POST(jsonString);

  Serial.printf("HTTP Response Code: %d\n", httpCode);

  if (httpCode == 200 || httpCode == 201) {
    lastServerStatus = "Success (HTTP " + String(httpCode) + ")";
    serverConnected  = true;
    lastServerSend   = millis();
    Serial.println("Server update SUCCESS");
  } else if (httpCode < 0) {
    lastServerStatus = "Connection error: " + http.errorToString(httpCode);
    serverConnected  = false;
    Serial.println("Server error: " + http.errorToString(httpCode));
  } else {
    lastServerStatus = "Server error: HTTP " + String(httpCode);
    serverConnected  = false;
    Serial.println("Server returned error: " + String(httpCode));
    Serial.println("Response: " + http.getString());
  }

  http.end();
  Serial.println("========================================");
  return serverConnected;
}

// Water level sensor
WaterLevelData readWaterLevel() {
  WaterLevelData result = {0, 0, false};
  if (Serial1.available() >= 4) {
    for (int i = 0; i < 4; i++) sensorData[i] = Serial1.read();
    if (sensorData[0] == 0xff) {
      int sum = (sensorData[0] + sensorData[1] + sensorData[2]) & 0x00FF;
      if (sum == sensorData[3]) {
        distance = (sensorData[1] << 8) + sensorData[2];
        if (distance > 280) {
          result.sensorReading = distance / 10.0;
          result.waterDepth    = sensorHeight - result.sensorReading;
          result.isValid       = true;
        }
      }
    }
    while (Serial1.available() > 0) Serial1.read();
  }
  return result;
}

// Save handlers
void handleSaveSTA() {
  if (server.hasArg("sta_ssid") && server.arg("sta_ssid").length() > 0) {
    saveWiFiCredentials(server.arg("sta_ssid"), server.arg("sta_password"), "");
    WiFi.disconnect();
    delay(500);
    initializeWiFi();
    server.sendHeader("Location", "/wifi-config?sta_saved=1");
    server.send(303);
    return;
  }
  server.sendHeader("Location", "/wifi-config");
  server.send(303);
}

void handleSaveAP() {
  if (server.hasArg("ap_ssid") && server.hasArg("ap_password")) {
    String newSSID = server.arg("ap_ssid");
    String newPass = server.arg("ap_password");
    if (newSSID.length() > 0 && newPass.length() >= 8) {
      saveWiFiCredentials("", "", newPass);
      ap_ssid = newSSID.c_str();
      WiFi.softAPdisconnect(true);
      delay(500);
      WiFi.softAP(ap_ssid, ap_password.c_str());
      server.sendHeader("Location", "/wifi-config?ap_saved=1");
      server.send(303);
      return;
    }
  }
  server.sendHeader("Location", "/wifi-config");
  server.send(303);
}

void handleSaveServer() {
  if (server.hasArg("server_host") && server.hasArg("server_path") &&
      server.hasArg("server_port") && server.hasArg("update_interval")) {
    String newHost      = server.arg("server_host");
    String newPath      = server.arg("server_path");
    int    newPort      = server.arg("server_port").toInt();
    unsigned long newInterval = server.arg("update_interval").toInt() * 1000;
    if (newHost.length() > 0 && newPath.length() > 0 && newPath.charAt(0) == '/' &&
        newPort > 0 && newPort <= 65535 && newInterval >= 5000 && newInterval <= 86400000) {
      saveServerConfig(newHost, newPath, newPort, newInterval);
      server.sendHeader("Location", "/wifi-config?server_saved=1");
      server.send(303);
      return;
    }
  }
  server.sendHeader("Location", "/wifi-config");
  server.send(303);
}

void handleOTAUpload() {
  HTTPUpload& upload = server.upload();

  if (upload.status == UPLOAD_FILE_START) {
    Serial.println("========================================");
    Serial.printf("OTA Update Start: %s\n", upload.filename.c_str());
    Serial.println("========================================");
    if (!Update.begin(UPDATE_SIZE_UNKNOWN)) {
      Update.printError(Serial);
    }
  }
  else if (upload.status == UPLOAD_FILE_WRITE) {
    if (Update.write(upload.buf, upload.currentSize) != upload.currentSize) {
      Update.printError(Serial);
    } else {
      // ✅ FIX: guard against division by zero when size is unknown
      if (Update.size() > 0) {
        Serial.printf("Progress: %d%%\r", (Update.progress() * 100) / Update.size());
      } else {
        Serial.printf("Written: %u bytes\r", Update.progress());
      }
    }
  }
  else if (upload.status == UPLOAD_FILE_END) {
    if (Update.end(true)) {
      Serial.printf("\nOTA Success: %u bytes written\n", upload.totalSize);
      Serial.println("Rebooting...");
      Serial.println("========================================");
      delay(500);
      ESP.restart();
    } else {
      Update.printError(Serial);
    }
  }
}

// Setup
void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("\n========================================");
  Serial.println("Water Depth & 4-Sensor Soil Moisture System");
  Serial.println("Device ID: " + String(DEVICE_ID));
  Serial.println("Firmware:  " + String(FIRMWARE_VERSION));
  Serial.println("========================================");

  EEPROM.begin(EEPROM_SIZE);

  if (EEPROM.read(ADDR_AP_PASS) == 255)
    saveWiFiCredentials("", "", "12345678");

  float savedHeight = EEPROM.readFloat(ADDR_SENSOR_HEIGHT);
  if (!isnan(savedHeight) && savedHeight > 0) sensorHeight = savedHeight;

  readWiFiCredentials();
  readAPPassword();
  readServerConfig();

  // ✅ FIX: empty completion handler — handleOTAUpload sends its own response (restart)

  Serial.println("OTA available at: /ota-update");
  Serial.println("========================================");

// Main loop
void loop() {
  server.handleClient();
  checkWiFiConnection();

  static unsigned long lastSensorRead  = 0;
  static unsigned long lastStatusPrint = 0;

  if (millis() - lastSensorRead >= 1000) {
    lastSensorRead = millis();
    WaterLevelData waterData = readWaterLevel();
    readSoilMoisture();
    getFormattedTimestamp(timeStampBuf, sizeof(timeStampBuf));

    if (wifiSTAConnected && !serverHost.isEmpty() && !serverPath.isEmpty() && serverPort > 0)
      serverConnected = sendToServer(waterData.waterDepth);
  }

  if (millis() - lastStatusPrint >= 5000) {
    lastStatusPrint = millis();
    WaterLevelData waterData = readWaterLevel();
    Serial.printf("\n=== System Status [%s] ===\n", timeStampBuf);
    Serial.printf("Device: %s\n", DEVICE_ID);
    Serial.printf("Water: sensor=%.1fcm depth=%.1fcm\n", waterData.sensorReading, waterData.waterDepth);
    for (int i = 0; i < NUM_SOIL_SENSORS; i++)
      Serial.printf("Soil %d: raw=%d %d%%\n", i+1, soilRaw[i], soilPercent[i]);
    Serial.printf("WiFi STA: %s | RSSI: %d dBm\n", wifiSTAConnected ? "Connected" : "Disconnected", WiFi.RSSI());
    Serial.printf("Server: %s | %s\n", serverConnected ? "Connected" : "Disconnected", lastServerStatus.c_str());
    Serial.printf("==============================\n");
  }

  delay(10);
}
