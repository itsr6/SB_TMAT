void initializeTime() {
  if (wifiSTAConnected) {
    configTime(gmtOffset_sec, daylightOffset_sec, ntpServer1, ntpServer2);
    
    time_t now = 0;
    struct tm timeinfo;
    int retry = 0;
    const int maxRetry = 10;
    
    while (time(&now) < 1000000000 && retry < maxRetry) {
      delay(500);
      retry++;
    }
    
    if (retry < maxRetry) {
      timeInitialized = true;
      lastTimeSync = millis();
      getLocalTime(&timeinfo);
      Serial.println("Time initialized successfully!");
    }
  }
}

bool getFormattedTimestamp(char* buffer, size_t bufferSize) {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    strcpy(buffer, "0000-00-00_00:00:00");
    return false;
  }
  strftime(buffer, bufferSize, "%d-%m-%Y_%H:%M:%S", &timeinfo);
  return true;
}

unsigned long getCurrentEpochTime() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    return 0;
  }
  time_t now;
  time(&now);
  return (unsigned long)now;
}

void readWiFiCredentials() {
  char ssid_buf[MAX_CRED_LENGTH] = {0};
  char pass_buf[MAX_CRED_LENGTH] = {0};
  
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    ssid_buf[i] = EEPROM.read(ADDR_WIFI_SSID + i);
    pass_buf[i] = EEPROM.read(ADDR_WIFI_PASS + i);
  }
  
  sta_ssid = String(ssid_buf);
  sta_password = String(pass_buf);
  sta_ssid.trim();
  sta_password.trim();
}

void readAPPassword() {
  char ap_pass_buf[MAX_CRED_LENGTH] = {0};
  bool valid = false;
  
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    ap_pass_buf[i] = EEPROM.read(ADDR_AP_PASS + i);
    if (ap_pass_buf[i] != 0 && ap_pass_buf[i] != 255) valid = true;
  }
  
  ap_password = String(ap_pass_buf);
  ap_password.trim();
  
  if (!valid || ap_password.isEmpty()) {
    ap_password = "12345678";
    for (int i = 0; i < ap_password.length(); i++) {
      EEPROM.write(ADDR_AP_PASS + i, ap_password[i]);
    }
    EEPROM.commit();
  }
}

void saveWiFiCredentials(String ssid, String password, String ap_password) {
  // Clear existing data
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_SSID + i, 0);
    EEPROM.write(ADDR_WIFI_PASS + i, 0);
    if (!ap_password.isEmpty()) {
      EEPROM.write(ADDR_AP_PASS + i, 0);
    }
  }
  
  // Write new credentials
  for (int i = 0; i < ssid.length() && i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_SSID + i, ssid[i]);
  }
  
  for (int i = 0; i < password.length() && i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_PASS + i, password[i]);
  }

  if (!ap_password.isEmpty()) {
    for (int i = 0; i < ap_password.length() && i < MAX_CRED_LENGTH; i++) {
      EEPROM.write(ADDR_AP_PASS + i, ap_password[i]);
    }
  }
  
  EEPROM.commit();
  sta_ssid = ssid;
  sta_password = password;
}

void initializeWiFi() {
  Serial.println("Initializing WiFi...");
  WiFi.mode(WIFI_AP_STA);
  
  if (!WiFi.softAP(ap_ssid, ap_password.c_str())) {
    WiFi.softAP(ap_ssid, "12345678");
  }

  if (sta_ssid.length() > 0) {
    WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    WiFi.setTxPower(WIFI_POWER_8_5dBm);
    
    unsigned long startTime = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - startTime < 10000) {
      delay(500);
      Serial.print(".");
    }

    if (WiFi.status() == WL_CONNECTED) {
      wifiSTAConnected = true;
      Serial.println("\nWiFi connected!");
      Serial.print("IP address: ");
      Serial.println(WiFi.localIP());
      initializeTime();
    }
  }
}

void checkWiFiConnection() {
  static unsigned long lastCheck = 0;
  const unsigned long checkInterval = 30000;
  
  if (millis() - lastCheck >= checkInterval) {
    lastCheck = millis();
    
    if (WiFi.status() != WL_CONNECTED) {
      Serial.println("WiFi disconnected, attempting to reconnect...");
      wifiSTAConnected = false;
      WiFi.reconnect();
      
      unsigned long startTime = millis();
      while (WiFi.status() != WL_CONNECTED && millis() - startTime < 10000) {
        delay(500);
        Serial.print(".");
      }
      
      if (WiFi.status() == WL_CONNECTED) {
        wifiSTAConnected = true;
        Serial.println("\nWiFi reconnected!");
        if (!timeInitialized) {
          initializeTime();
        }
      } else {
        Serial.println("\nWiFi reconnection failed");
      }
    }
  }
}

bool sendToServer(float waterDepth) {
  if (!wifiSTAConnected) {
    lastServerStatus = "WiFi not connected";
    Serial.println("Cannot send to server: WiFi not connected");
    return false;
  }

  // Check if it's time to send (every 20 seconds)
  if (millis() - lastServerAttempt < SERVER_UPDATE_INTERVAL) {
    return false;
  }

  lastServerAttempt = millis();

  Serial.println("========================================");
  Serial.println("Preparing to send data to server...");
  Serial.print("Server: ");
  Serial.println(serverHost);
  Serial.print("Port: ");
  Serial.println(serverPort);
  Serial.print("Path: ");
  Serial.println(serverPath);

  WiFiClientSecure client;
  client.setInsecure(); // Skip certificate verification
  
  Serial.print("Connecting to server... ");
  
  if (!client.connect(serverHost, serverPort)) {
    Serial.println("FAILED");
    lastServerStatus = "Connection failed";
    return false;
  }
  
  Serial.println("SUCCESS");
  
  // Get timestamp
  bool timeValid = getFormattedTimestamp(timeStampBuf, sizeof(timeStampBuf));
  
  // Create JSON document
  DynamicJsonDocument doc(512);
  
  // Add water depth data
  doc["device_id"] = DEVICE_ID;
  doc["sensor_reading"] = distance;
  doc["water_depth"] = waterDepth;
  doc["sensor_height"] = sensorHeight;
  doc["timestamp"] = timeStampBuf;
  doc["epoch_time"] = getCurrentEpochTime();
  
  // Serialize JSON to string
  String jsonString;
  serializeJson(doc, jsonString);
  
  Serial.println("JSON Data to send:");
  Serial.println(jsonString);
  
  // Prepare HTTP POST request
  String postRequest = String("POST ") + serverPath + " HTTP/1.1\r\n" +
                      "Host: " + serverHost + "\r\n" +
                      "Content-Type: application/json\r\n" +
                      "Content-Length: " + jsonString.length() + "\r\n" +
                      "Connection: close\r\n\r\n" +
                      jsonString;
  
  Serial.println("Sending HTTP request...");
  Serial.println(postRequest.substring(0, 200) + "..."); // Print first 200 chars
  
  // Send request
  client.print(postRequest);
  
  // Wait for response
  unsigned long timeout = millis();
  bool responseReceived = false;
  String response = "";
  int responseCode = 0;
  
  Serial.println("Waiting for response...");
  
  while (client.connected() && millis() - timeout < 10000) {
    while (client.available()) {
      String line = client.readStringUntil('\n');
      response += line + "\n";
      responseReceived = true;
      
      // Parse HTTP status code
      if (line.startsWith("HTTP/1.1")) {
        int space1 = line.indexOf(' ');
        int space2 = line.indexOf(' ', space1 + 1);
        if (space1 > 0 && space2 > space1) {
          responseCode = line.substring(space1 + 1, space2).toInt();
        }
      }
    }
  }
  
  if (responseReceived) {
    Serial.println("=== Server Response ===");
    Serial.println("HTTP Response Code: " + String(responseCode));
    Serial.println("Response Headers/Body:");
    Serial.println(response);
    Serial.println("======================");
    
    // Check if response contains success indication
    if (responseCode == 200 || response.indexOf("success") > 0 || response.indexOf("Success") > 0) {
      lastServerSend = millis();
      lastServerStatus = "Success (HTTP " + String(responseCode) + ")";
      serverConnected = true;
      Serial.println("Server update SUCCESS");
    } else {
      lastServerStatus = "Server error: HTTP " + String(responseCode);
      serverConnected = false;
      Serial.println("Server update FAILED");
    }
  } else {
    Serial.println("No response from server (timeout)");
    lastServerStatus = "No response from server";
    serverConnected = false;
  }
  
  client.stop();
  Serial.println("Connection closed");
  Serial.println("========================================");
  
  return serverConnected;
}

WaterLevelData readWaterLevel() {
  WaterLevelData result = {0, 0, false};
  
  if (Serial1.available() >= 4) {
    for (int i = 0; i < 4; i++) {
      sensorData[i] = Serial1.read();
    }
    
    if (sensorData[0] == 0xff) {
      int sum = (sensorData[0] + sensorData[1] + sensorData[2]) & 0x00FF;
      
      if (sum == sensorData[3]) {
        distance = (sensorData[1] << 8) + sensorData[2];
        
        if (distance > 280) {
          result.sensorReading = distance / 10.0;
          result.waterDepth = (result.sensorReading * -1) + sensorHeight;
          result.isValid = true;
          Serial.print("Valid sensor reading: ");
          Serial.print(result.sensorReading);
          Serial.print(" cm, Depth: ");
          Serial.print(result.waterDepth);
          Serial.println(" cm");
        }
      }
    }
    
    while (Serial1.available() > 0) Serial1.read();
  }
  
  return result;
}

void handleSaveSTA() {
  if (server.hasArg("sta_ssid") && server.hasArg("sta_password")) {
    String newSSID = server.arg("sta_ssid");
    String newPass = server.arg("sta_password");
    
    if (newSSID.length() > 0) {
      saveWiFiCredentials(newSSID, newPass, "");
      WiFi.disconnect();
      delay(1000);
      initializeWiFi();
      server.sendHeader("Location", "/wifi-config?sta_saved=1");
      server.send(303);
      return;
    }
  }
  server.sendHeader("Location", "/wifi-config");
  server.send(303);
}

void handleConfigure() {
  if (server.hasArg("sensorHeight")) {
    float newHeight = server.arg("sensorHeight").toFloat();
    if (newHeight != sensorHeight && newHeight > 0) {
      sensorHeight = newHeight;
      EEPROM.writeFloat(ADDR_SENSOR_HEIGHT, sensorHeight);
      EEPROM.commit();
      lastConfigUpdate = millis();
    }
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n========================================");
  Serial.println("Water Depth Monitoring System");
  Serial.println("Device ID: " + String(DEVICE_ID));
  Serial.println("========================================");
  
  EEPROM.begin(EEPROM_SIZE);
  if (EEPROM.read(ADDR_AP_PASS) == 255) {
    saveWiFiCredentials("", "", "12345678");
  }
  
  float savedHeight = EEPROM.readFloat(ADDR_SENSOR_HEIGHT);
  if (!isnan(savedHeight) && savedHeight > 0) {
    sensorHeight = savedHeight;
  }
  
  readWiFiCredentials();
  readAPPassword();
  
  Serial1.begin(9600, SERIAL_8N1, RX_PIN, -1);
  initializeWiFi();
  
  server.on("/", HTTP_GET, handleRoot);
  server.on("/readings", HTTP_GET, []() { server.send(200, "text/plain", generateReadingsHtml()); });
  server.on("/configure", HTTP_POST, handleConfigure);
  server.on("/wifi-config", HTTP_GET, handleWiFiConfig);
  server.on("/save-sta", HTTP_POST, handleSaveSTA);
  server.on("/save-ap", HTTP_POST, handleSaveAP);
  server.begin();

  Serial.println("Web server started");
  Serial.print("AP SSID: ");
  Serial.println(ap_ssid);
  Serial.print("AP IP: ");
  Serial.println(WiFi.softAPIP());
  Serial.println("System initialization complete");
  Serial.println("========================================");
}

void loop() {
  server.handleClient();
  checkWiFiConnection();

  static unsigned long lastSensorRead = 0;
  static unsigned long lastStatusPrint = 0;
  
  if (millis() - lastSensorRead >= 1000) {
    lastSensorRead = millis();
    WaterLevelData waterData = readWaterLevel();
    
    // Always try to send every 20 seconds if we have WiFi
    if (wifiSTAConnected) {
      serverConnected = sendToServer(waterData.waterDepth);
    }
  }

  if (millis() - lastStatusPrint >= 5000) { // Print status every 5 seconds
    lastStatusPrint = millis();
    WaterLevelData waterData = readWaterLevel();
    
    Serial.printf("\n=== System Status ===\n");
    Serial.printf("Device ID: %s\n", DEVICE_ID);
    Serial.printf("Sensor: %.1fcm | Depth: %.1fcm\n", waterData.sensorReading, waterData.waterDepth);
    Serial.printf("STA: %s | RSSI: %d dBm\n",
      wifiSTAConnected ? "Connected" : "Disconnected",
      WiFi.RSSI());
    Serial.printf("AP Clients: %d\n", WiFi.softAPgetStationNum());
    Serial.printf("Server: %s\n", serverConnected ? "Connected" : "Disconnected");
    Serial.printf("Last Server Status: %s\n", lastServerStatus.c_str());
    
    if (lastServerSend > 0) {
      unsigned long secondsAgo = (millis() - lastServerSend) / 1000;
      Serial.printf("Last send: %lu seconds ago\n", secondsAgo);
    }
    
    Serial.printf("Time: %s\n", timeStampBuf);
    Serial.printf("===================\n");
  }

  delay(10);
}
