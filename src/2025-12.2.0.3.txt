void initializeTime() {
  if (wifiSTAConnected) {
    configTime(gmtOffset_sec, daylightOffset_sec, ntpServer1, ntpServer2);
    
    time_t now = 0;
    struct tm timeinfo;
    int retry = 0;
    const int maxRetry = 10;
    
    while (time(&now) < 1000000000 && retry < maxRetry) {
      delay(500);
      retry++;
    }
    
    if (retry < maxRetry) {
      timeInitialized = true;
      lastTimeSync = millis();
      getLocalTime(&timeinfo);
      Serial.println("Time initialized successfully!");
    }
  }
}

bool getFormattedTimestamp(char* buffer, size_t bufferSize) {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    strcpy(buffer, "0000-00-00_00:00:00");
    return false;
  }
  strftime(buffer, bufferSize, "%d-%m-%Y_%H:%M:%S", &timeinfo);
  return true;
}

unsigned long getCurrentEpochTime() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    return 0;
  }
  time_t now;
  time(&now);
  return (unsigned long)now;
}

void readWiFiCredentials() {
  char ssid_buf[MAX_CRED_LENGTH] = {0};
  char pass_buf[MAX_CRED_LENGTH] = {0};
  
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    ssid_buf[i] = EEPROM.read(ADDR_WIFI_SSID + i);
    pass_buf[i] = EEPROM.read(ADDR_WIFI_PASS + i);
  }
  
  sta_ssid = String(ssid_buf);
  sta_password = String(pass_buf);
  sta_ssid.trim();
  sta_password.trim();
}

void readAPPassword() {
  char ap_pass_buf[MAX_CRED_LENGTH] = {0};
  bool valid = false;
  
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    ap_pass_buf[i] = EEPROM.read(ADDR_AP_PASS + i);
    if (ap_pass_buf[i] != 0 && ap_pass_buf[i] != 255) valid = true;
  }
  
  ap_password = String(ap_pass_buf);
  ap_password.trim();
  
  if (!valid || ap_password.isEmpty()) {
    ap_password = "12345678";
    for (int i = 0; i < ap_password.length(); i++) {
      EEPROM.write(ADDR_AP_PASS + i, ap_password[i]);
    }
    EEPROM.commit();
  }
}

void readUpdateInterval() {
  // Read 4 bytes for unsigned long
  unsigned long interval = 0;
  for (int i = 0; i < 4; i++) {
    byte b = EEPROM.read(ADDR_UPDATE_INTERVAL + i);
    interval |= ((unsigned long)b << (i * 8));
  }
  
  if (interval == 0xFFFFFFFF || interval < 1000 || interval > 86400000) {
    // Invalid or out of range, use default
    serverUpdateInterval = DEFAULT_UPDATE_INTERVAL;
    saveUpdateInterval(serverUpdateInterval);
  } else {
    serverUpdateInterval = interval;
  }
  
  Serial.print("Server update interval: ");
  Serial.print(serverUpdateInterval / 1000);
  Serial.println(" seconds");
}

void readServerConfig() {
  char host_buf[MAX_SERVER_LENGTH] = {0};
  char path_buf[MAX_SERVER_LENGTH] = {0};
  
  for (int i = 0; i < MAX_SERVER_LENGTH; i++) {
    host_buf[i] = EEPROM.read(ADDR_SERVER_HOST + i);
    path_buf[i] = EEPROM.read(ADDR_SERVER_PATH + i);
  }
  
  serverHost = String(host_buf);
  serverPath = String(path_buf);
  serverHost.trim();
  serverPath.trim();
  
  // Read port
  serverPort = EEPROM.read(ADDR_SERVER_PORT) | (EEPROM.read(ADDR_SERVER_PORT + 1) << 8);
  
  // Read update interval
  readUpdateInterval();  // This will set serverUpdateInterval
  
  // Check if configuration is valid
  bool configValid = true;
  
  if (serverHost.isEmpty() || serverHost.equals("0.0.0.0") || serverHost.indexOf('.') == -1) {
    configValid = false;
    Serial.println("Server host not configured or invalid");
  }
  
  if (serverPath.isEmpty() || serverPath.charAt(0) != '/') {
    configValid = false;
    Serial.println("Server path not configured or invalid");
  }
  
  if (serverPort <= 0 || serverPort > 65535) {
    configValid = false;
    Serial.println("Server port not configured or invalid");
  }
  
  if (!configValid) {
    Serial.println("Server configuration incomplete. Please configure via web interface.");
    lastServerStatus = "Not configured - Please set up server in web interface";
  } else {
    Serial.println("Server configuration loaded from EEPROM");
  }
}

void saveUpdateInterval(unsigned long interval) {
  // Save 4 bytes for unsigned long
  for (int i = 0; i < 4; i++) {
    EEPROM.write(ADDR_UPDATE_INTERVAL + i, (interval >> (i * 8)) & 0xFF);
  }
  serverUpdateInterval = interval;
}

void saveWiFiCredentials(String ssid, String password, String ap_password) {
  // Clear existing data
  for (int i = 0; i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_SSID + i, 0);
    EEPROM.write(ADDR_WIFI_PASS + i, 0);
    if (!ap_password.isEmpty()) {
      EEPROM.write(ADDR_AP_PASS + i, 0);
    }
  }
  
  // Write new credentials
  for (int i = 0; i < ssid.length() && i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_SSID + i, ssid[i]);
  }
  
  for (int i = 0; i < password.length() && i < MAX_CRED_LENGTH; i++) {
    EEPROM.write(ADDR_WIFI_PASS + i, password[i]);
  }

  if (!ap_password.isEmpty()) {
    for (int i = 0; i < ap_password.length() && i < MAX_CRED_LENGTH; i++) {
      EEPROM.write(ADDR_AP_PASS + i, ap_password[i]);
    }
  }
  
  EEPROM.commit();
  sta_ssid = ssid;
  sta_password = password;
}

void saveServerConfig(String host, String path, int port, unsigned long interval) {
  // Clear existing data
  for (int i = 0; i < MAX_SERVER_LENGTH; i++) {
    EEPROM.write(ADDR_SERVER_HOST + i, 0);
    EEPROM.write(ADDR_SERVER_PATH + i, 0);
  }
  
  // Write new config
  for (int i = 0; i < host.length() && i < MAX_SERVER_LENGTH; i++) {
    EEPROM.write(ADDR_SERVER_HOST + i, host[i]);
  }
  
  for (int i = 0; i < path.length() && i < MAX_SERVER_LENGTH; i++) {
    EEPROM.write(ADDR_SERVER_PATH + i, path[i]);
  }
  
  // Write port (2 bytes)
  EEPROM.write(ADDR_SERVER_PORT, port & 0xFF);
  EEPROM.write(ADDR_SERVER_PORT + 1, (port >> 8) & 0xFF);
  
  // Save update interval
  saveUpdateInterval(interval);
  
  EEPROM.commit();
  serverHost = host;
  serverPath = path;
  serverPort = port;
  serverUpdateInterval = interval;
  
  Serial.println("Server configuration saved:");
  Serial.print("Host: "); Serial.println(serverHost);
  Serial.print("Path: "); Serial.println(serverPath);
  Serial.print("Port: "); Serial.println(serverPort);
  Serial.print("Update Interval: "); Serial.print(interval / 1000); Serial.println(" seconds");
}

void initializeWiFi() {
  Serial.println("Initializing WiFi...");
  WiFi.mode(WIFI_AP_STA);
  
  if (!WiFi.softAP(ap_ssid, ap_password.c_str())) {
    WiFi.softAP(ap_ssid, "12345678");
  }

  if (sta_ssid.length() > 0) {
    WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    WiFi.setTxPower(WIFI_POWER_8_5dBm);
    
    unsigned long startTime = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - startTime < 10000) {
      delay(500);
      Serial.print(".");
    }

    if (WiFi.status() == WL_CONNECTED) {
      wifiSTAConnected = true;
      Serial.println("\nWiFi connected!");
      Serial.print("IP address: ");
      Serial.println(WiFi.localIP());
      initializeTime();
    }
  }
}

void checkWiFiConnection() {
  static unsigned long lastCheck = 0;
  const unsigned long checkInterval = 30000;
  
  if (millis() - lastCheck >= checkInterval) {
    lastCheck = millis();
    
    if (WiFi.status() != WL_CONNECTED) {
      Serial.println("WiFi disconnected, attempting to reconnect...");
      wifiSTAConnected = false;
      WiFi.reconnect();
      
      unsigned long startTime = millis();
      while (WiFi.status() != WL_CONNECTED && millis() - startTime < 10000) {
        delay(500);
        Serial.print(".");
      }
      
      if (WiFi.status() == WL_CONNECTED) {
        wifiSTAConnected = true;
        Serial.println("\nWiFi reconnected!");
        if (!timeInitialized) {
          initializeTime();
        }
      } else {
        Serial.println("\nWiFi reconnection failed");
      }
    }
  }
}

bool sendToServer(float waterDepth) {
  if (!wifiSTAConnected) {
    lastServerStatus = "WiFi not connected";
    Serial.println("Cannot send to server: WiFi not connected");
    return false;
  }

  // Check if server configuration is complete
  if (serverHost.isEmpty() || serverPath.isEmpty() || serverPort <= 0 || serverPort > 65535) {
    lastServerStatus = "Server not configured";
    Serial.println("Cannot send to server: Server configuration incomplete");
    return false;
  }

  if (millis() - lastServerAttempt < serverUpdateInterval) {
    return false;
  }

  lastServerAttempt = millis();

  Serial.println("========================================");
  Serial.println("Preparing to send data to server...");
  Serial.print("Server: ");
  Serial.println(serverHost);
  Serial.print("Port: ");
  Serial.println(serverPort);
  Serial.print("Path: ");
  Serial.println(serverPath);
  Serial.print("Update Interval: ");
  Serial.print(serverUpdateInterval / 1000);
  Serial.println(" seconds");

  WiFiClientSecure client;
  client.setInsecure();
  
  Serial.print("Connecting to server... ");
  
  if (!client.connect(serverHost.c_str(), serverPort)) {
    Serial.println("FAILED");
    lastServerStatus = "Connection failed";
    return false;
  }
  
  Serial.println("SUCCESS");
  
  bool timeValid = getFormattedTimestamp(timeStampBuf, sizeof(timeStampBuf));
  distance_cm = distance / 10.0;
  DynamicJsonDocument doc(512);
  
  doc["device_id"] = DEVICE_ID;
  doc["sensor_raw"] = distance_cm;
  doc["water_depth"] = waterDepth;
  doc["sensor_height"] = sensorHeight;
  doc["epoch_time"] = getCurrentEpochTime();
  
  String jsonString;
  serializeJson(doc, jsonString);
  
  Serial.println("JSON Data to send:");
  Serial.println(jsonString);
  
  String postRequest = String("POST ") + serverPath + " HTTP/1.1\r\n" +
                      "Host: " + serverHost + "\r\n" +
                      "Content-Type: application/json\r\n" +
                      "Content-Length: " + jsonString.length() + "\r\n" +
                      "Connection: close\r\n\r\n" +
                      jsonString;
  
  Serial.println("Sending HTTP request...");
  Serial.println(postRequest.substring(0, 200) + "...");
  
  client.print(postRequest);
  
  unsigned long timeout = millis();
  bool responseReceived = false;
  String response = "";
  int responseCode = 0;
  
  Serial.println("Waiting for response...");
  
  while (client.connected() && millis() - timeout < 10000) {
    while (client.available()) {
      String line = client.readStringUntil('\n');
      response += line + "\n";
      responseReceived = true;
      
      if (line.startsWith("HTTP/1.1")) {
        int space1 = line.indexOf(' ');
        int space2 = line.indexOf(' ', space1 + 1);
        if (space1 > 0 && space2 > space1) {
          responseCode = line.substring(space1 + 1, space2).toInt();
        }
      }
    }
  }
  
  if (responseReceived) {
    Serial.println("=== Server Response ===");
    Serial.println("HTTP Response Code: " + String(responseCode));
    Serial.println("Response Headers/Body:");
    Serial.println(response);
    Serial.println("======================");
    
    if (responseCode == 200 || response.indexOf("success") > 0 || response.indexOf("Success") > 0) {
      lastServerSend = millis();
      lastServerStatus = "Success (HTTP " + String(responseCode) + ")";
      serverConnected = true;
      Serial.println("Server update SUCCESS");
    } else {
      lastServerStatus = "Server error: HTTP " + String(responseCode);
      serverConnected = false;
      Serial.println("Server update FAILED");
    }
  } else {
    Serial.println("No response from server (timeout)");
    lastServerStatus = "No response from server";
    serverConnected = false;
  }
  
  client.stop();
  Serial.println("Connection closed");
  Serial.println("========================================");
  
  return serverConnected;
}

WaterLevelData readWaterLevel() {
  WaterLevelData result = {0, 0, false};
  
  if (Serial1.available() >= 4) {
    for (int i = 0; i < 4; i++) {
      sensorData[i] = Serial1.read();
    }
    
    if (sensorData[0] == 0xff) {
      int sum = (sensorData[0] + sensorData[1] + sensorData[2]) & 0x00FF;
      
      if (sum == sensorData[3]) {
        distance = (sensorData[1] << 8) + sensorData[2];
        
        if (distance > 280) {
          result.sensorReading = distance / 10.0;
          result.waterDepth = (result.sensorReading * -1) + sensorHeight;
          result.isValid = true;
          Serial.print("Valid sensor reading: ");
          Serial.print(result.sensorReading);
          Serial.print(" cm, Depth: ");
          Serial.print(result.waterDepth);
          Serial.println(" cm");
        }
      }
    }
    
    while (Serial1.available() > 0) Serial1.read();
  }
  
  return result;
}

void handleSaveServer() {
  if (server.hasArg("server_host") && server.hasArg("server_path") && 
      server.hasArg("server_port") && server.hasArg("update_interval")) {
    String newHost = server.arg("server_host");
    String newPath = server.arg("server_path");
    int newPort = server.arg("server_port").toInt();
    unsigned long newInterval = server.arg("update_interval").toInt() * 1000;  // Convert seconds to milliseconds
    
    // Validate inputs
    if (newHost.length() > 0 && 
        newPath.length() > 0 && newPath.charAt(0) == '/' &&
        newPort > 0 && newPort <= 65535 &&
        newInterval >= 5000 && newInterval <= 86400000) {  // 5 seconds to 24 hours
      
      saveServerConfig(newHost, newPath, newPort, newInterval);
      server.sendHeader("Location", "/wifi-config?server_saved=1");
      server.send(303);
      return;
    }
  }
  server.sendHeader("Location", "/wifi-config");
  server.send(303);
}

void handleOTAUpload() {
  HTTPUpload& upload = server.upload();
  
  if (upload.status == UPLOAD_FILE_START) {
    Serial.println("========================================");
    Serial.printf("OTA Update Start: %s\n", upload.filename.c_str());
    Serial.println("========================================");
    
    if (!Update.begin(UPDATE_SIZE_UNKNOWN)) {
      Update.printError(Serial);
      return;
    }
  } 
  else if (upload.status == UPLOAD_FILE_WRITE) {
    if (Update.write(upload.buf, upload.currentSize) != upload.currentSize) {
      Update.printError(Serial);
    } else {
      Serial.printf("Progress: %d%%\r", (Update.progress() * 100) / Update.size());
    }
  } 
  else if (upload.status == UPLOAD_FILE_END) {
    if (Update.end(true)) {
      Serial.println("\n========================================");
      Serial.printf("Update Success: %u bytes\n", upload.totalSize);
      Serial.println("Rebooting...");
      Serial.println("========================================");
      server.send(200, "text/plain", "Update successful! Rebooting...");
      delay(1000);
      ESP.restart();
    } else {
      Update.printError(Serial);
      server.send(500, "text/plain", "Update failed!");
    }
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n========================================");
  Serial.println("Water Depth Monitoring System");
  Serial.println("Device ID: " + String(DEVICE_ID));
  Serial.println("========================================");
  
  EEPROM.begin(EEPROM_SIZE);
  
  // Initialize default values if first run
  if (EEPROM.read(ADDR_AP_PASS) == 255) {
    saveWiFiCredentials("", "", "12345678");
  }
  
  float savedHeight = EEPROM.readFloat(ADDR_SENSOR_HEIGHT);
  if (!isnan(savedHeight) && savedHeight > 0) {
    sensorHeight = savedHeight;
  }
  
  readWiFiCredentials();
  readAPPassword();
  readServerConfig();  // This will load all server config including interval
  
  Serial1.begin(9600, SERIAL_8N1, RX_PIN, -1);
  initializeWiFi();
  
  // Start initial WiFi scan
  WiFi.scanNetworks(true);
  
  server.on("/", HTTP_GET, handleRoot);
  server.on("/readings", HTTP_GET, []() { server.send(200, "text/plain", generateReadingsHtml()); });
  server.on("/configure", HTTP_POST, handleConfigure);
  server.on("/wifi-config", HTTP_GET, handleWiFiConfig);
  server.on("/save-sta", HTTP_POST, handleSaveSTA);
  server.on("/save-ap", HTTP_POST, handleSaveAP);
  server.on("/save-server", HTTP_POST, handleSaveServer);
  
  // Add OTA routes
  server.on("/ota-update", HTTP_GET, handleOTAUpdate);
  server.on("/ota-upload", HTTP_POST, []() {
    server.send(200);
  }, handleOTAUpload);
  
  server.begin();

  Serial.println("Web server started");
  Serial.print("AP SSID: ");
  Serial.println(ap_ssid);
  Serial.print("AP IP: ");
  Serial.println(WiFi.softAPIP());
  
  Serial.println("\nServer Configuration Status:");

void loop() {
  server.handleClient();
  checkWiFiConnection();

  static unsigned long lastSensorRead = 0;
  static unsigned long lastStatusPrint = 0;
  
  if (millis() - lastSensorRead >= 1000) {
    lastSensorRead = millis();
    WaterLevelData waterData = readWaterLevel();
    
    if (wifiSTAConnected && !serverHost.isEmpty() && !serverPath.isEmpty() && serverPort > 0) {
      serverConnected = sendToServer(waterData.waterDepth);
    }
  }

  if (millis() - lastStatusPrint >= 5000) {
    lastStatusPrint = millis();
    WaterLevelData waterData = readWaterLevel();
    
    Serial.printf("\n=== System Status ===\n");
    Serial.printf("Device ID: %s\n", DEVICE_ID);
    Serial.printf("Sensor: %.1fcm | Depth: %.1fcm\n | Raw: %.1fcm\n", waterData.sensorReading, waterData.waterDepth, distance_cm);
    Serial.printf("STA: %s | RSSI: %d dBm\n",
      wifiSTAConnected ? "Connected" : "Disconnected",
      WiFi.RSSI());
    Serial.printf("AP Clients: %d\n", WiFi.softAPgetStationNum());
    
    // Server status
    if (serverHost.isEmpty() || serverPath.isEmpty() || serverPort <= 0) {
      Serial.printf("Server: NOT CONFIGURED\n");
    } else {
      Serial.printf("Server: %s | Status: %s\n", 
        serverConnected ? "Connected" : "Disconnected",
        lastServerStatus.c_str());
    }
    
    Serial.printf("Update Interval: %lu seconds\n", serverUpdateInterval / 1000);
    
    if (lastServerSend > 0) {
      unsigned long secondsAgo = (millis() - lastServerSend) / 1000;
      Serial.printf("Last send: %lu seconds ago\n", secondsAgo);
    }
    
    if (lastServerAttempt > 0) {
      unsigned long nextIn = (serverUpdateInterval - (millis() - lastServerAttempt)) / 1000;
      if (nextIn < serverUpdateInterval / 1000) {
        Serial.printf("Next send in: %lu seconds\n", nextIn);
      }
    }
    
    Serial.printf("Time: %s\n", timeStampBuf);
    Serial.printf("===================\n");
  }

  delay(10);
}
